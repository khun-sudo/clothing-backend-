            const storedItems = localStorage.getItem('hazel_master_items');

        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));

        function handleScan() {
            const inputField = document.getElementById('scan-input');
            const query = inputField.value.trim();
            if (!query) return;
            const item = masterItems.find(i => i.barcode === query || i.item_code.toLowerCase() === query.toLowerCase());
            if (item) { 
                if(item.stock_qty <= 0) { alert("⚠️ ဤပစ္စည်းသည် Stock လုံးဝမရှိတော့ပါ။"); }
                addToCart(item); 
                inputField.value = ""; 
            } else { alert("ပစ္စည်းရှာမတွေ့ပါ။"); }
            inputField.focus();
        }
        function addToCart(item) {
            const existing = cart.find(c => c.item_code === item.item_code);
            if (existing) existing.qty += 1;
            else cart.push({ ...item, qty: 1 });
            renderCart();
        }
        function updateCartQty(itemCode, newQty) {
            const item = cart.find(c => c.item_code === itemCode);
            if (item) { item.qty = parseInt(newQty) || 1; updateCartTotals(); }
        }
        function removeFromCart(itemCode) {
            cart = cart.filter(c => c.item_code !== itemCode);
            renderCart();
        }
        function renderCart() {
            const tbody = document.getElementById('cart-table-body');
            tbody.innerHTML = "";
            cart.forEach(item => {
                // Low stock validation alert style inside cart
                const isLow = (item.stock_qty - item.qty) < SHOP_CONFIG.lowStockThreshold;
                const warningTxt = isLow ? `<br><span class="text-[10px] text-red-500 font-bold">⚠️ Low Stock Alert (လက်ကျန်: ${item.stock_qty})</span>` : '';
                const tr = document.createElement('tr');
                tr.className = "border-b hover:bg-gray-50";
                tr.innerHTML = `<td class="p-2 font-medium">${item.item_name}${warningTxt}<br><span class="text-xs text-gray-400">${item.item_code}</span></td>
                    <td class="p-2">${item.sale_price_default.toLocaleString()}</td>
                    <td class="p-2"><input type="number" value="${item.qty}" min="1" class="w-16 border rounded text-center" onchange="updateCartQty('${item.item_code}', this.value)"></td>
                    <td class="p-2 font-semibold">${(item.sale_price_default * item.qty).toLocaleString()}</td>
                    <td class="p-2"><button onclick="removeFromCart('${item.item_code}')" class="text-red-500 font-bold">X</button></td>`;
                tbody.appendChild(tr);
            });
            updateCartTotals();
        }
        function updateCartTotals() {
            const currentCurrency = document.getElementById('currency-select').value;
            document.querySelectorAll('.currency-label').forEach(el => el.innerText = currentCurrency);
            let subtotal = cart.reduce((acc, item) => acc + (item.sale_price_default * item.qty), 0);
            let discount = parseFloat(document.getElementById('summary-discount').value) || 0;
            let total = Math.max(0, subtotal - discount);
            document.getElementById('summary-subtotal').innerText = subtotal.toLocaleString();
            document.getElementById('summary-total').innerText = total.toLocaleString();
        }

   function importCSV(type) {
            const fileInput = document.getElementById('csv-items-file');
            const file = fileInput.files[0];
            if (!file) return alert("CSV ဖိုင်အရင်ရွေးပါ။");
            const reader = new FileReader();
            reader.onload = function(e) {
                const text = e.target.result;
                const lines = text.split('\n');
                const headers = lines[0].split(',').map(h => h.trim().replace(/["']/g, ""));
                let importedData = [];
                for (let i = 1; i < lines.length; i++) {
                    if (!lines[i].trim()) continue;
                    const cols = lines[i].split(',').map(c => c.trim().replace(/["']/g, ""));
                    let obj = {};
                    headers.forEach((header, index) => { obj[header] = cols[index]; });
                    importedData.push(obj);
                }
                if (type === 'items') {
    masterItems = importedData.map(item => ({    
 item_code: item.item_code, barcode: item.barcode, item_name: item.item_name,
                        sale_price_default: parseFloat(item.sale_price_default) || 0, stock_qty: parseInt(item.stock_qty) || 0
                    }));
                    saveToStorage();
                    alert(`အောင်မြင်ပါသည်။ ပစ္စည်း ${masterItems.length} မျိုး သွင်းပြီးပါပြီ။`);
                    renderSetupItems();
                }
            };
            reader.readAsText(file);
        }

            dl.setAttribute("download", `hazel_sales_${new Date().toISOString().slice(0,10)}.json`);
